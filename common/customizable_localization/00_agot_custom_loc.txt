MaesterCheck = {
	type = character

	text = {
		trigger = {
			has_inactive_trait = grandmaester
			NOT = {
				has_trait = archmaester
			}
		}
		localization_key = grand_maester
	}

	text = {
		trigger = {
			has_trait = archmaester
			NOT = {
				has_inactive_trait = grandmaester
			}
		}
		localization_key = ArchaesterTitle
	}

	text = {
		trigger = {
			NOR = {
				has_trait = archmaester
				has_inactive_trait = grandmaester
			}
			OR = {
				AND = {
					has_trait = maester
					has_trait_xp = {
						trait = maester
						value = 100
					}
				}
				has_trait = maester_historical
			}
		}
		localization_key = MaesterTitle
	}

	text = {
		trigger = {
			NOT = { has_trait = archmaester }
			has_trait = maester
			has_trait_xp = {
				trait = maester
				value = 0
			}
		}
		localization_key = NoviceTitle
	}

	text = {
		localization_key = AcolyteTitle
	}
}
