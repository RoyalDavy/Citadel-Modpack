#SIGILS, TERRAIN, TITLES

GetWarriorNickname = {
	type = character
	random_valid = yes

	text = {
		trigger = { always = yes } #Generic
		localization_key = GetWarriorNickname_Blank  #The Wolf
	}
	text = {
		trigger = { exists = dynasty }
		localization_key = GetWarriorNickname_Dynasty  #The Wolf of Stark
	}
	text = {
		trigger = { is_landed = yes }
		localization_key = GetWarriorNickname_Location #The Wolf of Winterfell
	}
	text = {
		trigger = {
			is_agot_knight_trigger = yes
		}
		localization_key = GetWarriorNickname_Knight  #The Wolf Knight
	}
	text = {
		trigger = { has_trait = nightswatch }
		localization_key = GetWarriorNickname_Ranger  #The Wolf Ranger
	}
	text = {
		trigger = { culture = { has_cultural_pillar = heritage_ironborn } }
		localization_key = GetWarriorNickname_Reaver #The Wolf Reaver
	}
	text = {
		trigger = { is_female = yes }
		localization_key = GetWarriorNickname_Maid #The Wolf Maid
	}
	text = {
		trigger = {
			is_landed = yes
			highest_held_title_tier <= tier_kingdom
			agot_is_independent_ruler = no
		}
		localization_key = GetWarriorNickname_LordLady #The Wolf Lord
	}
	text = {
		trigger = {
			any_parent = {
				is_alive = yes
				highest_held_title_tier = tier_empire
			}
		}
		localization_key = GetWarriorNickname_PrincePrincess #The Wolf Prince
	}
	text = {
		trigger = {
			is_ruler = yes
			highest_held_title_tier = tier_empire
		}
		localization_key = GetWarriorNickname_KingQueen #The Wolf King
	}
	#Prince?
}
