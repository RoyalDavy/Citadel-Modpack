## Night's Watch Generic Positions ##

nw_septon = {
	max_available_positions = 1
	court_position_asset = {
		animation = prayer
		background = "gfx/interface/illustrations/event_scenes/the_wall.dds"
	}
	opinion = {
		value = regular_court_position_opinion
	}
	aptitude_level_breakpoints = {
		20
		40
		60
		80
	}
	aptitude = {
		value = 25
		add = {
			value = learning
			multiply = 2.5
			max = 50
			desc = court_position_skill_learning
		}
		if = {
			limit = {
				has_trait = zealous
			}
			add = {
				value = 20
				desc = court_position_zealous_trait
			}
		}
		if = {
			limit = {
				has_trait = compassionate
			}
			add = {
				value = 20
				desc = court_position_compassionate_trait
			}
		}
	}
	is_shown = {
		government_has_flag = government_is_nw
		agot_is_landless_nw_character = no
	}
	valid_position = {
		government_has_flag = government_is_nw
		agot_is_landless_nw_character = no
	}
	valid_character = {
		scope:employee = {
			is_male = yes
			is_adult = yes
			has_trait = nightswatch
			has_trait = septon
			base_court_position_validity_trigger = { EMPLOYER = scope:liege }
		}
	}
	revoke_cost = {
		prestige = {
			value = 200
			if = {
				limit = {
					exists = this
					is_ai = yes
				}
				multiply = 0
			}
		}
	}
	salary = {
		round = no
		gold = 0
		prestige = minor_court_position_prestige_salary
	}
	scaling_employer_modifiers = {
		terrible = {
			monthly_piety_gain_mult = 0.05
		}
		poor = {
			monthly_piety_gain_mult = 0.1
		}
		average = {
			monthly_piety_gain_mult = 0.15
		}
		good = {
			monthly_piety_gain_mult = 0.15
		}
		excellent = {
			monthly_piety_gain_mult = 0.2
		}
	}
	modifier = {
	}
	on_court_position_received = {
		basic_gained_court_position_effect = yes
	}
	on_court_position_revoked = {
		basic_revoked_court_position_effect = yes
	}
	on_court_position_invalidated = {
		basic_invalidated_court_position_effect = yes
	}
	candidate_score = {
		value = 50
		add = scope:base_value
		add = court_position_candidate_score_base_value
		add = court_position_candidate_aptitude_value
		scope:liege = {
			# Zealous AI's like this
			if = {
				limit = {
					ai_zeal > 75
				}
				add = 50
			}
		}
	}
}

b_the_shadow_tower_keeper = {
	max_available_positions = 1
	court_position_asset = {
		animation = marshall_random_weapon
		background = "gfx/interface/illustrations/event_scenes/the_wall.dds"
	}
	opinion = {
		value = regular_court_position_opinion
	}
	aptitude_level_breakpoints = {
		20
		40
		60
		80
	}
	aptitude = {
		value = 25
		add = {
			value = stewardship
			multiply = 1.25
			max = 25
			desc = court_position_skill_stewardship
		}
		add = {
			value = martial
			multiply = 1.25
			max = 25
			desc = court_position_skill_martial
		}
		if = {
			limit = {
				has_trait = administrator
			}
			add = {
				value = 20
				desc = court_position_administrator_trait
			}
		}
		if = {
			limit = {
				has_trait = overseer
			}
			add = {
				value = 20
				desc = court_position_overseer_trait
			}
		}
	}
	is_shown = {
		government_has_flag = government_is_nw
		OR = {
			any_vassal = {
				has_title = title:c_the_shadow_tower
			}
			has_title = title:c_the_shadow_tower
		}
		NOT = { primary_title = title:c_the_shadow_tower }
	}
	valid_position = {
		government_has_flag = government_is_nw
	}
	valid_character = {
		scope:employee = {
			OR = {
				is_landed = no
				has_title = title:c_the_shadow_tower
			}
			is_male = yes
			is_adult = yes
			has_trait = nightswatch
			NOR = {
				has_trait = maester
				has_trait = septon
			}
			base_court_position_validity_trigger = { EMPLOYER = scope:liege }
		}
	}
	revoke_cost = {
		prestige = {
			value = 200
			if = {
				limit = {
					exists = this
					is_ai = yes
				}
				multiply = 0
			}
		}
	}
	salary = {
		round = no
		gold = medium_court_position_salary
	}
	scaling_employer_modifiers = {
		terrible = {
			monthly_county_control_growth_add = 0.1
		}
		poor = {
			monthly_county_control_growth_add = 0.2
		}
		average = {
			monthly_county_control_growth_add = 0.3
		}
		good = {
			monthly_county_control_growth_add = 0.4
		}
		excellent = {
			monthly_county_control_growth_add = 0.5
		}
	}
	modifier = {
	}
	on_court_position_received = {
		scope:liege = {
			save_scope_as = employer_saved
		}
		scope:employee = {
			save_scope_as = employee_saved
		}
		create_title_and_vassal_change = {
			type = granted
			save_scope_as = title_change
			add_claim_on_loss = no
		}
		title:c_the_shadow_tower = {
			change_title_holder_include_vassals = {
				holder = scope:employee_saved
				change = scope:title_change
			}
		}
		resolve_title_and_vassal_change = scope:title_change
		scope:employee = {
			change_government = nights_watch_government
		}
	}
	on_court_position_revoked = {
		seneschal_title_revoked_effect = yes
		agot_on_shadowtower_court_position_revoked = yes
	}
	on_court_position_invalidated = {
		seneschal_title_revoked_effect = yes
		agot_on_shadowtower_court_position_revoked = yes
	}
	candidate_score = {
		value = 250
		add = scope:base_value
		add = court_position_candidate_score_base_value
		if = {
			limit = {
				NOT = {
					exists = scope:firing_court_position
				}
				scope:employee = {
					has_any_court_position = yes
				}
			}
			add = 150
		}
		if = {
			limit = {
				scope:employee = {
					has_title = title:c_the_shadow_tower
				}
			}
			add = 2000			#to stop AI from revoking alot
		}
		if = {
			limit = {
				scope:liege = {
					has_relation_rival = scope:employee
				}
			}
			add = 500			# rival isn't that big of a deal
		}
	}
}
