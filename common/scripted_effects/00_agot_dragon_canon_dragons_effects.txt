agot_spawn_canon_cradle_dragon_effect = {
	$CRADLEMATE$ = { save_scope_as = cradlemate }
	if = {
		limit = {
			scope:cradlemate = { is_character_targaryen_75 = yes }
		}
		agot_spawn_vermax_effect = yes
	}
	else_if = {
		limit = {
			scope:cradlemate = { is_character_targaryen_76 = yes }
		}
		agot_spawn_arrax_effect = yes
	}
	else_if = {
		limit = {
			scope:cradlemate = { is_character_targaryen_77 = yes }
		}
		agot_spawn_tyraxes_effect = yes
	}
	else_if = {
		limit = {
			scope:cradlemate = { is_character_targaryen_78 = yes }
		}
		agot_spawn_stormcloud_effect = yes
	}
	else_if = {
		limit = {
			scope:cradlemate = { is_character_targaryen_69 = yes }
		}
		agot_spawn_moondancer_effect = yes
	}
	else_if = {
		limit = {
			scope:cradlemate = { is_character_targaryen_70 = yes }
		}
		if = {
			limit = { has_character_flag = hatched_wyrmling }
			agot_spawn_morning_effect = yes
		}
		else = {
			agot_spawn_wyrmling_effect = yes
		}
	}
	else_if = {
		limit = {
			scope:cradlemate = { is_character_targaryen_9 = yes }
		}
		agot_spawn_drogon_effect = yes
	}
	else = {
		create_character = {
			location = scope:cradlemate.location
			template = agot_dragon_hatchling_template
			random_traits = no
			save_scope_as = dragon
		}
	}
}
