castellan_default_task_diplomacy_scale = {
	add = diplomacy
	if = {
		limit = {
			scope:councillor_liege = {
				#has_perk = loyalty_and_respect_perk
				culture = { has_cultural_parameter = close_family_better_councillors }
			}
			is_close_family_of = scope:councillor_liege
		}
		multiply = 1.1
	}

	divide = 5
	if = {
		limit = {
			scope:councillor_liege = {
				#has_perk = loyalty_and_respect_perk
				culture = { has_cultural_parameter = close_family_better_councillors }
			}
			is_close_family_of = scope:councillor_liege
		}
		ceiling = yes
	}
	else = {
		floor = yes
	}
	min = 1
}
