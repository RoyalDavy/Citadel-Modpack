agot_admiral_position_valid = {
	scope = character
	is_shown = {
		any_sub_realm_county = {
			any_county_province = {
				is_coastal = yes
			}
		}
	}
}

agot_spouse_position_valid = {
	scope = character
	is_shown = {
		NOT = {
			government_has_flag = government_is_nw
			has_diarchy_active_parameter = vizier_sits_in_the_council_instead_of_spouse
			has_character_flag = nw_ranger_title
		}
	}
}

agot_chaplain_position_valid = {
	scope = character
	is_shown = {
		NOT = {
			government_has_flag = government_is_nw
			has_character_flag = nw_ranger_title
		}
	}
}
