agot_dragon_skin_color_menu = {

		has_trait = dragon

				limit = {

# gene_dragon_primary_color_value +/-, reset
agot_dragon_skin_color_plus = {
	scope = character
	saved_scopes = {dragon}
	is_valid = {
		title:c_ruins.holder = {
			any_owned_story = {
				story_type = story_dragon_variable_storage
				var:dragon_id ?= root.var:editing_dragon
				trigger_if = {
					limit = {
						root.var:value_to_change = 0.01
						var:gene_dragon_primary_color_value < 1
					}
					var:gene_dragon_primary_color_value.agotadd001 <= 1.00
				}
				trigger_else_if = {
					limit = {
						root.var:value_to_change = 0.05
						var:gene_dragon_primary_color_value < 1
					}
					var:gene_dragon_primary_color_value <= 0.95
				}
				trigger_else = {
					always = no
				}
			}
		}
	}

	effect = {
		if = {
			limit = {
				agot_ruins_has_dragon_storage_system_editor_trigger = yes
			}
			title:c_ruins.holder = {
				every_owned_story = {
					limit = {
						story_type = story_dragon_variable_storage
						var:dragon_id ?= root.var:editing_dragon
					}
					save_scope_as = story
				}
				scope:story = {
					change_variable = {
						name = gene_dragon_primary_color_value
						add = root.agotaddvalue_to_change
					}
				}
			}
		}
	}
}
