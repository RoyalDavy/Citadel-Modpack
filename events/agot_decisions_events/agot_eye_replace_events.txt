# Event that removes your gem and reverts to the pirate patch
eyereplace.0005 = {
	type = character_event
	title = eyereplace.0001.t # Reuses the same loc
	desc = eyereplace.0005.desc
	theme = stewardship

	override_background = {
		reference = market
	}
	left_portrait = root

	option = {
		name = remove_facemask.0003.a
		remove_character_flag = has_eyegem_red
		remove_character_flag = has_eyegem_saphire
		remove_character_flag = has_eyegem_white
		remove_character_flag = has_eyegem_blue
		remove_character_flag = has_eyegem_green
		remove_character_flag = has_eyegem_black
		remove_character_flag = has_eyegem_yellow
		remove_character_flag = has_eyegem_orange
		remove_character_flag = has_eyegem_purple
	}
	option = {
		name = buy_facemask.0001.f
		remove_decision_cooldown = remove_eye_gem_decision
	}
}
