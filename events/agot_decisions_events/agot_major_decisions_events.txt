namespace = agot_major_decisions_events

agot_major_decisions_events.1000 = {
	hidden = yes

	immediate = {
		title:b_highgarden.holder = {
			if = {
				limit = {
					any_character_artifact = {
						has_variable = ruined_oakenseat
					}
				}
				trigger_event = agot_major_decisions_events.1001
			}
		}
	}
}

agot_major_decisions_events.1001 = {
	type = character_event
	title = agot_major_decisions_events.1001.t
	desc = agot_major_decisions_events.1001.desc
	theme = default
	left_portrait = {
		character = root
		animation = personality_content
	}
	right_portrait = {
		character = scope:greenhand
		animation = admiration
	}

	immediate = {
		random_character_artifact = {
			limit = { has_variable = ruined_oakenseat }
			destroy_artifact = this
		}
		agot_create_artifact_restored_oakenseat_effect = { OWNER = this }
		scope:newly_created_artifact = {
			equip_artifact_to_owner_replace = yes
		}
		character:Gardener_1 = {
			save_scope_as = greenhand
		}
	}

	option = {
		name = agot_major_decisions_events.1001.a
		root.dynasty = { add_dynasty_prestige = major_dynasty_prestige_value }
	}
}

	}
