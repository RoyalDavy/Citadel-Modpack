	type = character_event

	override_background = {
		reference = courtyard
	}

	right_portrait = {
		character = root
		animation = happiness
	}

		is_available = yes
		is_landed = yes
		effective_age > 6
		any_held_county = {
			title_province = {
				geographical_region = world_westeros
				NOT = {
					geographical_region = world_westeros_beyond_the_wall
				}
			}
		}
		NOT = {
			has_character_flag = agot_filler_hedges_puppeteer
		}
	}

	immediate = {
		add_character_flag = {
			flag = agot_filler_hedges_puppeteer
			years = 5
		}
	}

	option = { #adult watching puppet show
		name = agot_filler_hedges.0001.a
		trigger = {
			effective_age >= 14
		}

			random_list = {
				50 = {
					trigger_event = agot_filler_hedges.0002

					trigger_event = agot_filler_hedges.0003
				}
			}
		}
	}

	option = { #child watching puppet show

			effective_age < 14
		}
		hidden_effect = {
			random_list = {

					trigger_event = agot_filler_hedges.0002
