#start the event
agot_filler_crannog.0001 = {
	type = character_event
	title = agot_filler_crannog.0001.t
	desc = agot_filler_crannog.0001.desc

	theme = hunt_activity

	override_background = {
		reference = wilderness_wetlands
	}

	left_portrait = {
		character = root
		animation = scheme
	}

	right_portrait = {
		character = scope:agot_filler_foreigner_stuck_in_the_neck_npc_scope
		animation = fear
		hide_info = yes
	}

	trigger = {
		any_held_county = {
			title_province = { geographical_region = world_westeros_the_neck }
		}
		has_culture = culture:crannogman #does crannog heritage exist/will exist? otherwise need a way to determine crannog culture for divergences and hybrids.
		is_landed = yes
		is_available_adult = yes
		NOT = { has_character_flag = agot_filler_foreigners_stuck_in_the_neck }
	}

	immediate = {
		add_character_flag = agot_filler_foreigners_stuck_in_the_neck
		create_character = {
			location = root.capital_province
			culture = culture:riverlander
			faith = faith:fots_seven
			gender = male
			save_scope_as = agot_filler_foreigner_stuck_in_the_neck_npc_scope
		}
	}

	#Help them and get them out of *my* home sooner.
	option = {
		name = agot_filler_crannog.0001.a
		trigger_event = {
			id = agot_filler_crannog.0002
			days = 1
		}
		stress_impact = {
			sadistic = major_stress_gain
			compassionate = major_stress_loss
			generous = major_stress_loss
		}
	}

	#Leave them. The Neck deals with such problems on its own.
	option = {
		name = agot_filler_crannog.0001.b
		add_dread = minor_dread_gain
		if = {
			limit = {
				number_of_personality_traits < personality_trait_limit
				NOR = {
					has_trait = callous
					has_trait = compassionate
				}
			}
			random = {
				chance = 65
				add_trait = callous
			}
		}
		stress_impact = {
			sadistic = major_stress_loss
			lazy = minor_stress_loss
			callous = major_stress_loss
			compassionate = massive_stress_gain
			generous = major_stress_gain
		}
	}

	#Help them, feed them, and escort them the rest of the way.
	option = {
		name = agot_filler_crannog.0001.c
		trigger = {	has_trait = compassionate }
		trait = compassionate
		trigger_event = {
			id = agot_filler_crannog.0003
			days = 1
		}
		stress_impact = {
			sadistic = massive_stress_gain
			lazy = miniscule_stress_impact_gain
			callous = massive_stress_gain
			compassionate = massive_stress_loss
			generous = major_stress_loss
		}
	}

	#Bait some lizard-lions to there location and enjoy the show.
	option = {
		name = agot_filler_crannog.0001.d
		trigger = {	has_trait = sadistic }
		trait = sadistic
		trigger_event = {
			id = agot_filler_crannog.0004
			days = 1
		}
		stress_impact = {
			sadistic = massive_stress_loss
			lazy = miniscule_stress_impact_gain
			callous = massive_stress_loss
			compassionate = massive_stress_gain
			generous = massive_stress_gain
		}
	}
}

agot_filler_crannog.0009 = {
	type = character_event
	title = agot_filler_crannog.0009.t
	desc = agot_filler_crannog.0009.desc

	theme = hunt_activity

	left_portrait = {
		character =	root
		animation = aggressive_sword
	}

	override_background = {
		reference = wilderness_wetlands
	}

	option = {
		name = agot_filler_crannog.0009.a #fight it

		duel = {
			skill = prowess
			value = extremely_high_skill_rating
			# You kill and secure the Lizard-lion
			50 = {
				compare_modifier = {
					value = scope:duel_value #compare the prowess diff
					# so x's prowess vs extremely_high_skill_rating
					multiplier = 3.5 #apply a multiplier to... value
					min = -49
				}
				desc = agot_filler_crannog.0009.a.tt_success
				send_interface_toast = {
					title = agot_filler_crannog.0009.a.tt_success
					left_icon = root
					# Per the comment below the immediate block, the static gold is deliberate.
					add_gold = 55
					add_prestige = minor_prestige_gain
				}
			}
			# You are mauled, but live....
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				desc = agot_filler_crannog.0009.a.tt_failure_mauled
				send_interface_toast = {
					title = agot_filler_crannog.0009.a.tt_failure_mauled
					left_icon = root
					add_prestige = minor_prestige_loss
					add_trait = wounded_3

				}
			}
			# You Die! Ouch!
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -4.5
					min = -49
				}
				desc = agot_filler_crannog.0009.a.tt_failure_death
				send_interface_toast = {
					title = agot_filler_crannog.0009.a.tt_failure_death
					left_icon = root
					root = {
						death = {
							death_reason = death_ripped_to_death_by_lizard_lion
						}
					}

				}
			}
		}
	}

	#On second thought... risk it off-path.
	option = {
		name = agot_filler_crannog.0009.b
		trigger_event = {
			id = agot_filler_crannog.0010
			days = 1
		}
	}

}

agot_filler_crannog.0014 = {
	type = character_event
	title = agot_filler_crannog.0014.t
	desc = agot_filler_crannog.0014.desc

	theme = martial_strategy_focus

	left_portrait = {
		character = root
		animation = thinking
	}

	override_background = {
		reference = wilderness_wetlands
	}

	trigger = {
		OR = {
			AND = {
				is_available_adult = yes
				any_held_county = {
					title_province = { geographical_region = world_westeros_the_neck }
				}
			}
			agot_filler_crannog_army_in_the_neck_trigger = yes
		}
		NOT = { has_character_flag = agot_filler_find_old_crannog_spear }
		exists = scope:location
		exists = scope:location.title_province
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			OR = {
				has_trait = lifestyle_blademaster
			}
		}
		modifier = {
			add = 0.5
			OR = {
				has_trait = education_martial_4
				has_trait = education_martial_3
			}
		}
		modifier = {
			add = -0.5
			has_trait = craven
		}
	}

	immediate = {
		add_character_flag = {
			flag = agot_filler_find_old_crannog_spear
		}
	}

	#Get the artifact
	option = {
		name = agot_filler_crannog.0014.a
		create_artifact = {
			name = agot_filler_crannog_artifact.0014.name
			description = agot_filler_crannog_artifact.0014.desc
			visuals = spear
			type = spear
			modifier = artifact_prowess_1_modifier
			wealth = 5
			quality = 15
			save_scope_as = newly_created_artifact
			history = {
				location = scope:location.title_province
				recipient = root
				type = discovered
			}
		}
		stress_impact = {
			greedy = major_stress_impact_loss
		}
	}

	#leave it
	option = {
		name = agot_filler_crannog.0013.b
		stress_impact = {
			greedy = major_stress_impact_gain
		}
	}

}
