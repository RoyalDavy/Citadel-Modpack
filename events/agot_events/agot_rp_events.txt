scripted_trigger westeros_descriptions = {
	any_held_county = {
		title_province = {
			geographical_region = world_westeros_seven_kingdoms
		}
	}
}
scripted_trigger not_westeros_descriptions = {
	NOR = {
		any_held_county = {
			title_province = {
				geographical_region = world_westeros_seven_kingdoms
			}
		}
		any_held_county = {
			title_province = {
				geographical_region = world_westeros_beyond_the_wall
			}
		}
	}
}
agot_rp_event.0001 = { #start
	type = character_event
	title = agot_rp_event.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					westeros_descriptions = yes
				}
				desc = agot_rp_event.0001.desc_westeros
			}
			desc = agot_rp_event.0001.desc
		}
	}

	theme = dynasty
	override_background = {
		reference = throne_room
	}

	right_portrait = {
		character = scope:rp_maester
		animation = throne_room_writer
	}

	immediate = {
		play_music_cue = "mx_cue_succession_instrumental"
		hidden_effect = {
			if = {
				limit = {
					global_var:citadel_title.holder = {
						any_courtier_or_guest = {
							has_trait_xp = {
								trait = maester
								value = 100
							}
							NOR = {
								has_trait = archmaester
								has_character_flag = maester_researcher
							}
							is_adult = yes
						}
					}
				}
				global_var:citadel_title.holder = {
					random_courtier_or_guest = {
						limit = {
							has_trait_xp = {
								trait = maester
								value = 100
							}
							NOR = {
								has_trait = archmaester
								has_character_flag = maester_researcher
							}
							is_adult = yes
						}
						save_scope_as = rp_maester
						if = {
							limit = {
								has_character_flag = blocked_from_leaving
							}
							remove_character_flag = blocked_from_leaving
						}
					}
				}
			}
			else = {
				create_character = {
					template = maester_character
					location = root.capital_province
					dynasty = none
					save_scope_as = rp_maester
				}
			}
			scope:rp_maester = {
				add_character_flag = traveling_maester
			}
		}
	}

	option = { #I'll tell my tale
		name = agot_rp_event.0001.a
		if = {
			limit = {
				westeros_descriptions = yes
			}
			custom_tooltip = agot_rp_event.0001.a.tt_westeros
		}
		else = {
			custom_tooltip = agot_rp_event.0001.a.tt
		}

		trigger = {
			NOR = {
				has_culture = culture:high_valyrian
				has_culture = culture:westerosi_valyrian
			}
		}
		trigger_event = agot_rp_event.0002
	}

	option = { #I'll tell my tale
		name = agot_rp_event.0001.b
		if = {
			limit = {
				westeros_descriptions = yes
			}
			custom_tooltip = agot_rp_event.0001.a.tt_westeros
		}
		else = {
			custom_tooltip = agot_rp_event.0001.a.tt
		}
		trigger = {
			OR = {
				has_culture = culture:high_valyrian
				has_culture = culture:westerosi_valyrian
			}
		}
		trigger_event = agot_rp_event.0010
	}

	# Maester, please record my dynastic appearance
	option = {
		name = agot_rp_event.0001.d
		custom_tooltip = agot_rp_event.0001.d.tt
		custom_tooltip = agot_rp_event.0001.d.tt_warning
		trigger_event = agot_rp_event.0100
	}

	option = { #begone maester
		name = {
			text = agot_rp_event.0001.c
		}
		name = {
			text = agot_rp_event.0001.c_westeros
			trigger = {
				westeros_descriptions = yes
			}
		}
		custom_tooltip = agot_rp_event.0001.c.tt
		hidden_effect = {
			scope:rp_maester ?= {
				save_scope_as = maester
				send_traveling_maester_to_citadel = yes
			}
		}
	}
}
