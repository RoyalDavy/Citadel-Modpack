								culture = scope:secret_owner.var:orig_culture

									AND = {
										is_male = yes

										is_female = yes

		}
		name = {
			trigger = {
				opinion = {
					target = scope:owner

				}

			TARGET = scope:owner

		}

			base = 100

				ai_zeal = low_chance_impact_positive_ai_value
			}
			modifier = {
				has_trait = zealous
				add = 100
			}
			modifier = {
				opinion = {
					target = scope:owner

				}
				add = 100

				OR = {
					is_close_family_of = scope:owner

		}
	}

	#For liege: forgive
	option = {
		trigger = {
			allowed_to_imprison_character_trigger = { CHARACTER = scope:owner }
			culture = { agot_is_wildling_culture = no }
		}

			target = scope:owner

			opinion = 30
		}

			zealous = medium_stress_impact_gain
		}
